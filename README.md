# vue-wheel-number
vue component of wheel number

# install
 ```bash
 $ npm i vue-wheel-number -s
 ```
# demo
```
 /example
```
# use npm
```js
  import Vue from 'Vue'
  import WheelNumber from 'vue-wheel-number'
  Vue.use(WheelNumber)
  
  //or
  import {WheelNumber,WheelUpdateNumber} from 'vue-wheel-number'
  export default {
    components: {
      WheelNumber,WheelUpdateNumber
    }
  }
```
